STARTUP
1. git pull origin main
	-to check if any changes have been made by others and to update your local repo to current version

2. git status
	-check for any differences in file versions

WORKFLOW
1. git checkout -b insert-branch-feature-name
	-change to new branch for creating new project feature

2. git add .
	-add all files changes to the staging area for commit

3. git commit -m "insert descriptive message here"
	-commit change to local repo branch

4. git push origin branch-name
	-to push branch to remote for merge request to main repo


OTHERS
1. git checkout main
	-switch back to main branch

2. git push origin main
	-push main branch to remote repo
